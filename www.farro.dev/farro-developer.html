# Python — Multi-thread Port Scanner
import socket, threading
from datetime import datetime

def scan_port(host, port, results):
    try:
        s = socket.socket()
        s.settimeout(0.5)
        if s.connect_ex((host, port)) == 0:
            try: svc = socket.getservbyport(port)
            except: svc = 'unknown'
            results.append(f"[+] {port:5}/tcp  OPEN  ({svc})")
        s.close()
    except: pass

def scan(host, start=1, end=1024):
    results, threads = [], []
    print(f"\n[*] Scanning {host} | {datetime.now():%H:%M:%S}")
    for p in range(start, end+1):
        t = threading.Thread(target=scan_port,args=(host,p,results))
        t.start(); threads.append(t)
    [t.join() for t in threads]
    [print(r) for r in sorted(results)]
    print(f"[*] Done — {len(results)} open ports found")

scan("127.0.0.1")
// JavaScript — XSS Prevention & Secure Fetch
const Security = {

  // Sanitize HTML to prevent XSS attacks
  sanitize(str) {
    const map = {
      '&':'&amp;', '<':'&lt;', '>':'&gt;',
      '"':'&quot;', "'":'&#x27;'
    };
    return str.replace(/[&<>"']/g, m => map[m]);
  },

  // Generate secure CSRF token
  csrfToken() {
    const arr = new Uint8Array(32);
    crypto.getRandomValues(arr);
    return [...arr].map(b=>b.toString(16).padStart(2,'0')).join('');
  },

  // Safe POST request with CSRF header
  async post(url, data) {
    const res = await fetch(url, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'X-CSRF-Token': Security.csrfToken()
      },
      body: JSON.stringify(data)
    });
    if (!res.ok) throw new Error(res.statusText);
    return res.json();
  }
};
